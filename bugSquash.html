const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); const swatter = document.getElementById('swatter'); let gameRunning = true; let score = 0; let timeLeft = 60; let bugs = []; let powerUps = []; let particles = []; let combo = 0; let comboTimer = 0; let isFrozen = false; let freezeTimer = 0; let multiplier = 1; let multiplierTimer = 0; const bugTypes = [ { emoji: 'üêõ', speed: 1.5, points: 10, size: 30 }, { emoji: 'üêû', speed: 2, points: 15, size: 25 }, { emoji: 'ü¶ü', speed: 3, points: 20, size: 20 }, { emoji: 'ü™≤', speed: 1, points: 25, size: 35 }, { emoji: 'üï∑Ô∏è', speed: 2.5, points: 30, size: 28 }, { emoji: 'ü¶Ç', speed: 1.8, points: 40, size: 32 } ]; const powerUpTypes = [ { emoji: '‚ùÑÔ∏è', type: 'freeze', color: '#00bfff' }, { emoji: '‚ú®', type: 'multiplier', color: '#ffd700' } ]; // Mouse tracking for swatter let mouseX = canvas.width / 2; let mouseY = canvas.height / 2; canvas.addEventListener('mousemove', (e) => { const rect = canvas.getBoundingClientRect(); mouseX = e.clientX - rect.left; mouseY = e.clientY - rect.top; swatter.style.left = e.clientX + 'px'; swatter.style.top = e.clientY + 'px'; }); canvas.addEventListener('mousedown', () => { swatter.classList.add('squashing'); checkBugSquash(); }); canvas.addEventListener('mouseup', () => { swatter.classList.remove('squashing'); }); // Spawn bugs function spawnBug() { const type = bugTypes[Math.floor(Math.random() * bugTypes.length)]; const edge = Math.floor(Math.random() * 4); let x, y; switch(edge) { case 0: x = Math.random() * canvas.width; y = -type.size; break; case 1: x = canvas.width + type.size; y = Math.random() * canvas.height; break; case 2: x = Math.random() * canvas.width; y = canvas.height + type.size; break; case 3: x = -type.size; y = Math.random() * canvas.height; break; } bugs.push({ ...type, x, y, vx: (Math.random() - 0.5) * type.speed, vy: (Math.random() - 0.5) * type.speed, rotation: Math.random() * Math.PI * 2, rotationSpeed: (Math.random() - 0.5) * 0.1, changeDirectionTimer: Math.random() * 60 + 30, alive: true }); } // Spawn power-ups function spawnPowerUp() { const type = powerUpTypes[Math.floor(Math.random() * powerUpTypes.length)]; powerUps.push({ ...type, x: Math.random() * (canvas.width - 40) + 20, y: Math.random() * (canvas.height - 40) + 20, size: 30, life: 5000, // 5 seconds pulse: 0 }); } // Create particles function createParticles(x, y, color, count = 15) { for (let i = 0; i < count; i++) { particles.push({ x, y, vx: (Math.random() - 0.5) * 8, vy: (Math.random() - 0.5) * 8, size: Math.random() * 6 + 2, color, life: 1, gravity: 0.2 }); } } // Check if bug is squashed function checkBugSquash() { let squashed = false; bugs.forEach(bug => { if (!bug.alive) return; const dist = Math.sqrt((bug.x - mouseX) ** 2 + (bug.y - mouseY) ** 2); if (dist < bug.size) { bug.alive = false; score += bug.points * multiplier; combo++; comboTimer = 60; createParticles(bug.x, bug.y, '#8b4513', 20); squashed = true; // Send win message if (window.parent) { window.parent.postMessage({ type: 'game-result', result: 'win' }, '*'); } } }); // Check power-ups powerUps.forEach((powerUp, index) => { const dist = Math.sqrt((powerUp.x - mouseX) ** 2 + (powerUp.y - mouseY) ** 2); if (dist < powerUp.size) { if (powerUp.type === 'freeze') { isFrozen = true; freezeTimer = 180; // 3 seconds createParticles(powerUp.x, powerUp.y, '#00bfff', 25); } else if (powerUp.type === 'multiplier') { multiplier = 2; multiplierTimer = 300; // 5 seconds createParticles(powerUp.x, powerUp.y, '#ffd700', 25); } powerUps.splice(index, 1); } }); if (!squashed) { combo = 0; } } // Update bugs function updateBugs() { bugs = bugs.filter(bug => bug.alive && bug.x > -100 && bug.x < canvas.width + 100 && bug.y > -100 && bug.y < canvas.height + 100); bugs.forEach(bug => { if (!isFrozen) { bug.changeDirectionTimer--; if (bug.changeDirectionTimer <= 0) { bug.vx = (Math.random() - 0.5) * bug.speed * 2; bug.vy = (Math.random() - 0.5) * bug.speed * 2; bug.changeDirectionTimer = Math.random() * 60 + 30; } bug.x += bug.vx; bug.y += bug.vy; bug.rotation += bug.rotationSpeed; // Bounce off walls if (bug.x < 0 || bug.x > canvas.width) bug.vx *= -1; if (bug.y < 0 || bug.y > canvas.height) bug.vy *= -1; bug.x = Math.max(0, Math.min(canvas.width, bug.x)); bug.y = Math.max(0, Math.min(canvas.height, bug.y)); } }); // Spawn more bugs if (bugs.length < 8 + Math.floor(score / 200)) { if (Math.random() < 0.02) { spawnBug(); } } } // Update power-ups function updatePowerUps() { powerUps = powerUps.filter(p => { p.life -= 16.67; // ~60fps p.pulse += 0.1; return p.life > 0; }); // Spawn power-ups occasionally if (powerUps.length < 2 && Math.random() < 0.005) { spawnPowerUp(); } } // Update particles function updateParticles() { particles = particles.filter(p => { p.x += p.vx; p.y += p.vy; p.vy += p.gravity; p.vx *= 0.95; p.vy *= 0.95; p.life -= 0.02; return p.life > 0; }); } // Update timers function updateTimers() { if (comboTimer > 0) comboTimer--; if (combo <= 0 || comboTimer <= 0) combo = 0; if (freezeTimer > 0) { freezeTimer--; if (freezeTimer <= 0) isFrozen = false; } if (multiplierTimer > 0) { multiplierTimer--; if (multiplierTimer <= 0) multiplier = 1; } } // Draw bugs function drawBugs() { bugs.forEach(bug => { if (!bug.alive) return; ctx.save(); ctx.translate(bug.x, bug.y); ctx.rotate(bug.rotation); // Shadow ctx.shadowBlur = 10; ctx.shadowColor = 'rgba(0,0,0,0.3)'; ctx.font = `${bug.size}px Arial`; ctx.textAlign = 'center'; ctx.textBaseline = 'middle'; ctx.fillText(bug.emoji, 0, 0); ctx.restore(); }); } // Draw power-ups function drawPowerUps() { powerUps.forEach(powerUp => { const scale = 1 + Math.sin(powerUp.pulse) * 0.2; ctx.save(); ctx.translate(powerUp.x, powerUp.y); ctx.scale(scale, scale); // Glow effect ctx.shadowBlur = 15; ctx.shadowColor = powerUp.color; ctx.font = `${powerUp.size}px Arial`; ctx.textAlign = 'center'; ctx.textBaseline = 'middle'; ctx.fillText(powerUp.emoji, 0, 0); ctx.restore(); }); } // Draw particles function drawParticles() { particles.forEach(p => { ctx.globalAlpha = p.life; ctx.fillStyle = p.color; ctx.beginPath(); ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2); ctx.fill(); }); ctx.globalAlpha = 1; } // Draw UI function drawUI() { // Background for UI ctx.fillStyle = 'rgba(0,0,0,0.7)'; ctx.fillRect(0, 0, canvas.width, 60); ctx.fillStyle = '#fff'; ctx.font = 'bold 24px Arial'; ctx.textAlign = 'left'; ctx.fillText(`Score: ${score}`, 20, 35); ctx.textAlign = 'center'; ctx.fillStyle = timeLeft <= 10 ? '#ff3333' : '#fff'; ctx.fillText(`Time: ${timeLeft}s`, canvas.width / 2, 35); ctx.textAlign = 'right'; ctx.fillStyle = '#fff'; ctx.fillText(`Bugs: ${bugs.filter(b => b.alive).length}`, canvas.width - 20, 35); // Combo if (combo > 1) { ctx.textAlign = 'center'; ctx.fillStyle = '#ffd700'; ctx.font = 'bold 32px Arial'; ctx.strokeStyle = '#000'; ctx.lineWidth = 3; ctx.strokeText(`${combo}x COMBO!`, canvas.width / 2, 100); ctx.fillText(`${combo}x COMBO!`, canvas.width / 2, 100); } // Power-up indicators let indicatorY = 80; if (isFrozen) { ctx.fillStyle = 'rgba(0, 191, 255, 0.7)'; ctx.fillRect(10, indicatorY, 150, 30); ctx.fillStyle = '#fff'; ctx.font = 'bold 16px Arial'; ctx.textAlign = 'left'; ctx.fillText(`‚ùÑÔ∏è FREEZE: ${Math.ceil(freezeTimer / 60)}s`, 20, indicatorY + 20); indicatorY += 40; } if (multiplier > 1) { ctx.fillStyle = 'rgba(255, 215, 0, 0.7)'; ctx.fillRect(10, indicatorY, 150, 30); ctx.fillStyle = '#fff'; ctx.font = 'bold 16px Arial'; ctx.textAlign = 'left'; ctx.fillText(`‚ú® 2x SCORE: ${Math.ceil(multiplierTimer / 60)}s`, 20, indicatorY + 20); } // Instructions ctx.font = '14px Arial'; ctx.textAlign = 'center'; ctx.fillStyle = 'rgba(255,255,255,0.7)'; ctx.fillText('Click to squash bugs! Collect power-ups!', canvas.width / 2, canvas.height - 20); } // End game function endGame() { gameRunning = false; ctx.fillStyle = 'rgba(0,0,0,0.85)'; ctx.fillRect(0, 0, canvas.width, canvas.height); ctx.fillStyle = '#fff'; ctx.font = 'bold 48px Arial'; ctx.textAlign = 'center'; ctx.fillText('TIME\'S UP!', canvas.width / 2, canvas.height / 2 - 60); ctx.font = '32px Arial'; ctx.fillText(`Final Score: ${score}`, canvas.width / 2, canvas.height / 2); ctx.fillText(`Bugs Squashed: ${score / 10}`, canvas.width / 2, canvas.height / 2 + 50); ctx.font = '20px Arial'; ctx.fillStyle = '#aaa'; ctx.fillText('Refresh to play again', canvas.width / 2, canvas.height / 2 + 100); if (window.parent) { window.parent.postMessage({ type: 'game-end' }, '*'); } } // Game timer setInterval(() => { if (gameRunning) { timeLeft--; if (timeLeft <= 0) { endGame(); } } }, 1000); // Spawn initial bugs for (let i = 0; i < 5; i++) { spawnBug(); } // Game loop function gameLoop() { if (!gameRunning) return; // Clear const gradient = ctx.createLinearGradient(0, 0, 0, canvas.height); gradient.addColorStop(0, '#87ceeb'); gradient.addColorStop(1, '#e0f6ff'); ctx.fillStyle = gradient; ctx.fillRect(0, 0, canvas.width, canvas.height); // Draw grass pattern ctx.fillStyle = 'rgba(34, 139, 34, 0.3)'; for (let i = 0; i < 20; i++) { ctx.fillRect(0, canvas.height - 50 + i * 3, canvas.width, 2); } updateBugs(); updatePowerUps(); updateParticles(); updateTimers(); drawParticles(); drawPowerUps(); drawBugs(); drawUI(); requestAnimationFrame(gameLoop); } gameLoop();
